requirejs笔记

requirejs是一个js三方插件，支持Js的CMD和AMD规范。

##全局函数
requirejs提供的函数，使用require.toUrl()形式调用

  • toUrl(“app/dep.js”)
    得到该资源的完整URL，用来填充资源地址等
  • undef(‘jquery’)
    阻止依赖该模块的js继续加载，直至该模块加载完

使用

  1. 在页面中引入requirejs
    使用延迟加载，尽可能晚的加载requirejs，确保其使用的不符合规范的js文件能在其之前加载。
    data-main定义了requirejs的主文件，该文件定义了需要使用的js文件路径，别名，依赖关系等等。

    <script src="js/require.js" defer async="true" data-main="js/main"></script>

  2. 定义无依赖模块 - dep.js

    //无依赖模块定义，
//第一个参数为数组，表明依赖的模块，
//第二个参数为函数，返回任意类型数据，
define([], function(){
    function dep(){
        console.log('this is dep');
    }
    return {
        dep : dep
    };
});

  3. 定义AMD规范模块 - test.js

    //第一个参数为数组，表明依赖的模块，可以为main.js中定义的别名，也可以是相对于require.js的路径；
//第二个参数为函数，返回任意类型
//第三个参数为异常处理函数
define(['require','./app/dep'], function(require,dep){
    function foo(){
        console.log('this is foo');
        console.log('i will use dep');
        dep.dep();
    }
    return {
        foo : foo
    };
}, function (err) {
    //异常处理
    console.log(err);
    var failedId = err.requireModules && err.requireModules[0];
    if(failedId == 'app/dep'){}
});

  4. 定义CMD规范模块 - test_cmd.js

    //将require作为参数，使用其他模块时动态require
define(function(require, exports, module){
    function foo(){
        console.log('this is foo');
        console.log('i will use dep');
        //在这里动态加载
        require("./app/dep").dep();
    }
    return {
        foo : foo
    };
}, function (err) {
    //异常处理
    console.log(err);
    var failedId = err.requireModules && err.requireModules[0];
    if(failedId == 'app/dep'){}
});

  5. 定义requirejs主文件基本设置-main.js

    requirejs.config({
    // 基目录,与下面配置的路径有关,默认值为RequireJS所在目录
    // 模块查找地址一般为 baseUrl + paths，除非
    //    1.以 ".js" 结束        -其地址解析为对于当前HTML的相对地址
    //    2.以 "/" 开始,
    //    3.包含 URL 协议, 如 "http:" or "https:".
    baseUrl: "js/",

    // 配置模块名与对应文件
    paths: {
        // 可以使用网络地址
        "jquery": "https://ajax.googleapis.com/ajax/libs/jquery/1.7.2/jquery.min",
        // 可以传入数组容错，依次加载直至加载成功
        "jquery2": ["https://ajax.googleapis.com/ajax/libs/jquery/1.7.3/jquery.min",
                'lib/jquery'
            ],
        "test": "app/test",
        'jquery.scrollTo': "lib/jquery.scrollTo.js",
    },
    // 配置不符合AMD规范的库
    shim: {
　　　　'jquery.scrollTo': {
            // 定义依赖的模块，这些模块应该在加载scroll之前就被加载
　　　　　　deps: ['jquery'],
            // 定义对外提供调用的对象名，调用scroll来使用该模块
　　　　　　exports: 'scrollTo',
            // 初始化
            init: function (jquery) {
                // 依赖会作为参数传入，进行某些处理
                // this指向jquery.scrollTo的全局变量
                // 如果具有返回值，返回值会覆盖exports设置的值
                return this.Foo.noConflict();
            }
　　　　}
　　},
    // 不同目录下的依赖将得到不同的js文件，提高代码向后兼容性
     map: {
         // 所有模块使用foo1.2.js，优先级最低，可以被下面的配置覆盖
        '*': {
            'foo': 'foo1.2'
        },
        // 对于some文件夹下的oldmodule模块使用foo1.0.js
        'some/oldmodule': {
            'foo': 'foo1.0'
        }
    },
    // 为模块设置某些值方便使用传递
    /*
    test.js
        define(function (require, exports, module) {
            //Will be the value 'large'
            var size = module.config().size;
        });
    or
        define(['module'], function (module) {
            //Will be the value 'large'
            var size = module.config().size;
        });
    */
      config: {
        'test': {
            size: 'large'
        },
    }
});

// 全局异常事件
requirejs.onError = function (err) {
    console.log(err);
    if (err.requireType === 'timeout') {
        console.log('modules: ' + err.requireModules);
    }
};

  6. 在主文件中使用domReady插件 -main.js
    插件文件domReady.js应该放在baseUrl目录下，默认为requirejs所在目录
    domReady有两种调用方式，阻塞与非阻塞，在声明依赖是进行区分
    非阻塞方式：

    //该方法是异步方法，不阻塞继续执行
require(['domReady'], function (domReady) {
  domReady(function () {
      console.log('dom is ready');
  });
});

    阻塞方式：

    //该方法会一直等待dom加载，直至dom加载完成执行函数，参数为document
//声明依赖时多了一个`!`
require(['domReady!'], function (doc) {
    require(['test'], function (test){
        test.foo();
    });
});
